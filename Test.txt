main() {
  double side1 = 1;
  int side2 = 2;
  int side3 = 3;

  /* Ingresar lados del triángulo */
  print("Ingrese lado 1: ");
  input(side1);
  print("Ingrese lado 2: ");
  input(side2);
  print("Ingrese lado 3: ");
  input(side3);
  
  if (side1 == side2 && side2 == side3){
    /*Todos los lados son iguales*/
    println("Triángulo Equilátero!!");
  } else {
    if (side1 ==side2 || side1 == side3 || (side2 == side3)){
        /*Hay dos lados iguales*/
        println("Triángulo iscóceles");
    } else {
        if (side1 != side2 && side1 != side3 && side2 != side3){
            /*No hay lados iguales*/
            println("Triángulo escaleno");
        }
    }
  }
}
